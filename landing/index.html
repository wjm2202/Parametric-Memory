quickstart

# one-line start
docker-compose up

# write an atom
curl -X POST localhost:3000/atoms \
  -d '{"atom":"v1.fact.user_prefers_dark_mode"}'

# access + get Markov prediction
curl -X POST localhost:3000/access \
  -d '{"atom":"v1.fact.user_prefers_dark_mode"}'

# → returns atom + predicted_next + proof.verified: true
